class KNN:
    def __init__(self,X,y,k):
        self.X=X.values
        self.y=y.values
        self.k=k
        self.num_class=np.unique(y.values).shape[0] #total number of (unique) classes
        self.classes=np.unique(y.values) #the classes (unique)
        self.class_count=dict()
        for i in self.classes:
            self.class_count[i]=0  #number of points in each class is set to 0
        
    
    def predict(self,test):
        if(self.X.shape[1]!=test.shape[0]):   #check if dimensions of point to classify and those of X match
            print("Number of dimensions do not match..")
            return
        l=[]
        for i in range(self.X.shape[0]):
            x1=self.X[i][0]
            x2=self.X[i][1]
            dist=math.sqrt(((test[0]-x1)**2)+((test[1]-x2)**2))     #calculate distance between the point to classify and every other point '(x,y)'
            l.append([dist, self.y[i], self.X[i][0], self.X[i][1]])  # record the [distance bewteen (x,y) ad test point, class of the point (x,y), x, y]
        l.sort()
        self.topk=l[:self.k] #select the closest k points 
        for i in range(self.k):
            self.class_count[l[i][1]]+=1 #count the number of points in topk in each class 
        return max(self.class_count,key=lambda x: self.class_count[x]) #return the class with highest count
    
    def plot(self,df,point):
        plt.scatter(df[df["y"]==0]["x1"],df[df["y"]==0]["x2"],c='orange')
        plt.scatter(df[df["y"]==1]["x1"],df[df["y"]==1]["x2"],c='green')            
        plt.scatter(point[0],point[1],c='red')
        x,y=[point[0]],[point[1]]
        for i in self.topk:
            plt.plot([i[2],point[0]],[i[3],point[1]],'y-')
        plt.title("With k="+str(self.k))
        plt.show()

class WeightedKNN:
    def __init__(self,X,y,k):
        self.X=X.values
        self.y=y.values
        self.k=k
        self.num_class=np.unique(y.values).shape[0] #total number of (unique) classes
        self.classes=np.unique(y.values) #the classes (unique)
        self.class_count=dict()
        for i in self.classes:
            self.class_count[i]=0 #sum of weights of each class is set to 0
        
    def predict(self,test):
        if(self.X.shape[1]!=test.shape[0]):
            print("Number of dimensions do not match..")
            return
        l=[]
        tot_weight=0 #sum of weights of k closest points
        for i in range(self.X.shape[0]):
            x1=self.X[i][0]
            x2=self.X[i][1]
            dist=math.sqrt(((test[0]-x1)**2)+((test[1]-x2)**2))
            weight=1/dist                 #calculate the weight (inverse of distance)
            tot_weight+=weight            #adding to find total weight / sum of weights
            l.append([dist, weight, self.y[i], self.X[i][0], self.X[i][1]]) # record the [distance bewteen (x,y) and test point, weight, class of the point (x,y), x, y]
        l.sort()
        
        self.topk=l[:self.k] #select the closest k points
        
        for i in range(self.k):
            self.class_count[l[i][2]]+=(l[i][1]/tot_weight) #sum up the sum of (weight/total_weight) for each class separately 
            
        return max(self.class_count,key=lambda x: self.class_count[x]) #return the class with maximum sum
    
    def plot(self,df,point):
        plt.scatter(df[df["y"]==0]["x1"],df[df["y"]==0]["x2"],c='orange')
        plt.scatter(df[df["y"]==1]["x1"],df[df["y"]==1]["x2"],c='green')            
        plt.scatter(point[0],point[1],c='red')
        x,y=[point[0]],[point[1]]
        for i in self.topk:
            plt.plot([i[3],point[0]],[i[4],point[1]],'y-')
        plt.title("With k="+str(self.k))
        plt.show()
